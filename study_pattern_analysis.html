#df_count 만들고, 정렬을 위해 날짜 7월 3일 -> 7월 03일 변경
import re

df_count = s_count.to_frame(name='강의 수')
df_count = df_count.reset_index()
df_count.columns = ['날짜', '강의 수']

def change_num(date):
    date = re.sub('(\d)월 (\d)일', r'\1월 0\2일', date)
    return date
    
df_count['날짜'] = df_count['날짜'].map(change_num)
df_count
